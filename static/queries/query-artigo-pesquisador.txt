SELECT (COUNT(DISTINCT ?pesquisador) AS ?Pesquisadores) (COUNT(DISTINCT ?artigo) AS ?Artigos) WHERE {
  ?artigo wdt:P361 wd:Q18477654.
  
  {
    ?artigo wdt:P50 ?pesquisador.  # autor identificado
  }
  UNION
  {
    ?artigo wdt:P2093 ?nomeAutor.  # autor com nome literal (string)
    BIND(STR(?nomeAutor) AS ?pesquisador)  # trata como se fosse identificador
  }
}
